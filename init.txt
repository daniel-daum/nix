=====
Step 1: Install Xcode CLI tools
=====
xcode-select --install

=====
Step 2: Install Rosetta
=====
softwareupdate --install-rosetta

=====
Step 3: Install Determinate Nix
=====
curl --proto '=https' --tlsv1.2 -sSf -L \
  https://install.determinate.systems/nix | sh -s -- install

=====
Step 4: Install Determinate Nix
=====
git clone https://tangled.org/danieldaum.net/nix ~/.config/nix

=====
Step 5: Make Nix profile directory
=====
mkdir -p ~/.local/state/nix/profiles


=====
Step 6: First darwin rebuild
=====
cd ~/.config/nix && sudo nix run nix-darwin/master#darwin-rebuild -- switch --flake .

=====
Step 7: create .gitconfig.local
=====
echo '[user]
    signingkey = /Users/daniel/Library/Containers/com.maxgoedjen.Secretive.SecretAgent/Data/PublicKeys/<key-hash>.pub' > ~/.gitconfig.local

=====
Step 8: create .gitallowedsigners
=====
echo 'daniel@danieldaum.net ecdsa-sha2-nistp256 <public-key> <machine-name>@secretive.<hostname>.local' > ~/.gitallowedsigners

=====
Step 9: Add key to JJ
=====
jj config set --user signing.key "ecdsa-sha2-nistp256 <public-key>"
